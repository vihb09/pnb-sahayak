# Python add-on packages this project needs.
# Install them all with:  py -m pip install -r requirements.txt

# --- Live voice pipeline & core ---
requests        # makes the web calls to the Sarvam API
python-dotenv   # reads your secret key from the .env file

# --- Knowledge base (real PNB documents) ---
pypdf           # reads the text layer out of PDFs
sarvamai        # official Sarvam SDK, used for OCR of scanned PDFs (Document Digitization)
rank_bm25       # simple keyword search over the policy documents

# --- Web app (the voice assistant page + server) ---
fastapi          # the web framework that runs the server
uvicorn          # runs the FastAPI app locally
python-multipart # lets the server receive the uploaded microphone audio

# --- Live streaming ---
numpy            # resamples audio to 16 kHz for streaming
websockets       # real-time streaming WebSocket (also used by the Sarvam SDK)
